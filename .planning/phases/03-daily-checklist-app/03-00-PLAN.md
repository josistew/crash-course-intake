       ```typescript
       import type { Config } from 'jest';
       import nextJest from 'next/jest.js';

       const createJestConfig = nextJest({ dir: './' });

       const config: Config = {
         testEnvironment: 'jest-environment-jsdom',
         setupFilesAfterSetup: ['<rootDir>/jest.setup.ts'],
         moduleNameMapper: {
           '^@/(.*)$': '<rootDir>/src/$1',
         },
         testPathPattern: 'src/__tests__/',
       };

       export default createJestConfig(config);
       ```

       NOTE: Use `next/jest` wrapper (handles CSS/image transforms automatically). If Next.js version does not export `next/jest`, fall back to raw ts-jest config with manual moduleNameMapper for CSS/images.

       ```typescript
       import '@testing-library/jest-dom';
       ```

    4. Add test script to package.json if not present:
